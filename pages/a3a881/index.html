1.1 еҲӣе»әиЎЁ

from django.db import models

class UserInfo(models.Model):
    username = models.CharField(max_length=32)
    def __str__(self):
        return self.username

class UserGroup(models.Model):
    group_name = models.CharField(max_length=64)
    user_info = models.ManyToManyField(to='UserInfo',related_query_name='m2m')

    def __str__(self):
        return self.group_name

02.иҮӘе·ұеҲӣе»әз¬¬дёүеј е…ізі»иЎЁ(жі•2)

  • з¬¬дәҢз§Қпјҡ иҮӘе·ұеҲӣе»әз¬¬дёүеј е…ізі»иЎЁпјҢж—  m2m еӯ—ж®өпјҢиҮӘе·ұй“ҫиЎЁжҹҘиҜў
from django.db import models

#иЎЁ1пјҡдё»жңәиЎЁ
class Host(models.Model):
    nid = models.AutoField(primary_key=True)
    hostname = models.CharField(max_length=32,db_index=True)

#иЎЁ2пјҡеә”з”ЁиЎЁ
class Application(models.Model):
    name = models.CharField(max_length=32)

#иЎЁ3пјҡиҮӘе®ҡд№үз¬¬дёүеј е…іиҒ”иЎЁ
class HostToApp(models.Model):
    hobj = models.ForeignKey(to="Host",to_field="nid")
    aobj = models.ForeignKey(to='Application',to_field='id')

# еҗ‘з¬¬дёүеј иЎЁжҸ’е…Ҙж•°жҚ®пјҢе»әз«ӢеӨҡеҜ№еӨҡеӨ–й”®е…іиҒ”
HostToApp.objects.create(hobj_id=1,aobj_id=2)

    obj.m2m.remove(*[1, 2, 3])    # еҲ йҷӨз¬¬дёүеј иЎЁдёӯзҡ„пјҲ1,1пјүпјҲ1,2пјүпјҲ1,3пјүжқЎзӣ®
